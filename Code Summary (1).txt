# find association rules with default settings (Support: both A and B, doesn't matter the direction [redundant rules]; 
#                                               Confidence: shows how confident the algorithm is in regards to this association;
#                                               Lift: is a measure of performance of a targey model at classifying or predicting the cases.it is a ratio)
rules=apriori(bid.arule)

#---------------------------------------------------------------------------------------------------

rules<- apriori (bid.arule, parameter = list(minlen=2, supp=0.005, conf=0.8), 
                 appearance = list(rhs=c("BidStatus=WON", "BidStatus=LOST"),
                 default="lhs"),
                 control = list(verbose=F))


rules.sorted<- sort(rules, by="lift")
inspect(rules.sorted)
#create a dataframe:
df=data.frame(

#---------------------------------------------------------------------------------------------------
# Pruning Redundant Rules
# when a rule is a super rule of another rule, the former rule is considered to be redundant.
# find redundant rules
subset.matrix <- is.subset (rules.sorted, rules.sorted)
subset.matrix[lower.tri(subset.matrix, diag=T)] <- NA
redundant <- colSums(subset.matrix, na.rm=T)>= 1
which(redundant)
# remove redundant rules
rules.pruned <- rules.sorted[!redundant]
inspect (rules.pruned)        
rules.pruned <- rules[!is.redundant(rules.sorted)]
arules::inspect(rules.pruned)

lhs = labels(lhs(rules.pruned)),
rhs = labels(rhs(rules.pruned)),
rules.pruned@quality)

ash <- lumber.price [which(lumber.price$Species=="ASH"),]

ash

abline(reg=lm(ash~time(ash)))
cycle(ash)#This will print the cycle across years.
plot(aggregate(ash,FUN=mean))#This will aggregate the cycles and display a year on year trend
boxplot(ash~cycle(ash))#Box plot across months will give us a sense on seasonal effect


acf.ash <- acf (ash) # autocorrelation
pac.ash <- pacf (ash) # partial autocorrelation






# decomposing non-seasonal data
ash.sma<- SMA (ash, type="mult")

Sub SaveSheets()
    Dim strPath As String
    Dim ws As Worksheet


    Application.ScreenUpdating = False


    strPath = ActiveWorkbook.Path & "\"
    For Each ws In ThisWorkbook.Sheets
        ws.Copy
        'Use this line if you want to break any links:
        BreakLinks Workbooks(Workbooks.Count)
        Workbooks(Workbooks.Count).Close True, strPath & ws.Name & ".xlsx"
    Next


    Application.ScreenUpdating = True
End Sub


Sub BreakLinks(wb As Workbook)
    Dim lnk As Variant
    For Each lnk In wb.LinkSources(xlExcelLinks)
        wb.BreakLink lnk, xlLinkTypeExcelLinks
    Next
End Sub

Close the VBA editor

select 
        s.userID as "UserID"
        ,Auditlog.*
    ,cast(JSON_VALUE(s.data, '$.activated')as date) as "LastActivatedAt"
        ,JSON_VALUE(s.data, '$.promoCode') as "PromoCode"
        ,JSON_VALUE(s.data, '$.active') as "isCurrentActive"
        ,cast(JSON_VALUE(s.data, '$.created') as date) as "CreatedAt"
        ,cast(JSON_VALUE(s.data, '$.cancelled') as date) as "LastCancelledAt"
    ,JSON_VALUE(s.data, '$.braintreeID') as "CurrentBraintreeID"
        ,JSON_VALUE(s.data, '$.isPayingSubscriber') as "isPayingSubscriber"
        ,JSON_VALUE(s.data, '$.isZombie') as "isZombie"
        ,JSON_VALUE(s.data, '$.lastReconcile') as "LastReconcile"
        ,JSON_VALUE(s.data, '$.braintreeStatus') as "BraintreeStatus"
        ,JSON_VALUE(s.data, '$.paymentToken') as "PaymentToken"
        ,BT.*  
from [csidb].[csi].[Subscription] as s                                                                  /* subscription most recent record*/ 
full outer join                                                                                         /* since the three subscription tables are created at different time i use full outer join here to include all the records*/
    (
        select 
            a.id as AuditID
        ,JSON_VALUE(a.previousContent, '$.subID') as "SubscriptionID"
        ,JSON_VALUE(a.previousContent, '$.previousSubscriptionData.braintreeID') as "BraintreeID"
        ,JSON_VALUE(a.previousContent, '$.previousSubscriptionData.userID') as "UserID"
        ,cast(JSON_VALUE(a.previousContent, '$.previousSubscriptionData.lastRenewal')as date) as "ActivatedAt"
        ,case when JSON_VALUE(a.previousContent, '$.previousSubscriptionData.cancelled')>=JSON_VALUE(a.previousContent, '$.previousSubscriptionData.activated')           /* remove the cancel date when subscriber rejoin and active now*/
                  then cast(JSON_VALUE(a.previousContent, '$.previousSubscriptionData.cancelled') as date)
                      else null
            end as "CancelledAt"
            ,JSON_VALUE(previousContent, '$.newStatus') as "Status"
                ,JSON_VALUE(previousContent, '$.previousSubscriptionData.data.kickback') as "Kickback"
            ,JSON_VALUE(previousContent, '$.previousSubscriptionData.data.kickbackPriceThreshold') as "KickbackPriceThreshold"
            ,JSON_VALUE(previousContent, '$.previousSubscriptionData.data.price') as "Price"
            ,JSON_VALUE(previousContent, '$.previousSubscriptionData.data.lastKickbackPercentage') as "LastKickbackPercentage"


    from [csidb].[csi].[AuditLog] as a
            join                                                                 
          (select
    JSON_VALUE(previousContent, '$.previousSubscriptionData.braintreeID') as "BraintreeID"
              ,max(id) as max_id
          from 
              [csidb].[csi].[AuditLog]
          group by 
              JSON_VALUE(previousContent, '$.previousSubscriptionData.braintreeID')) as suba
          on 
              JSON_VALUE(a.previousContent, '$.previousSubscriptionData.braintreeID')=suba.BraintreeID 
              and a.id=suba.max_id


          WHERE
    a.id > 16208 and JSON_VALUE(a.previousContent, '$.previousSubscriptionData.braintreeID') <>' ' )  as Auditlog                                                   
on s.userID= Auditlog.UserID 
full outer join 
    (
    select                                                                                                 
        bt.id as WebhookID
            ,JSON_VALUE(bt.data, '$.subscriptionID') as Braintree_ID
            ,JSON_VALUE(bt.data, '$.userID') as User_ID
        ,JSON_VALUE(bt.data, '$.subscriptionStatus') as Subscription_Status
            ,JSON_VALUE(bt.data, '$.kind') as Kind
            ,JSON_VALUE(bt.data, '$.daysPastDue') as PastDue_Days
            ,JSON_VALUE(bt.data, '$.balance') as Balance
            ,JSON_VALUE(bt.data, '$.failureCount') as Failure_Count
            ,cast(JSON_VALUE(bt.data, '$.firstBillingDate')as date) as First_Billing_Date
            ,cast(JSON_VALUE(bt.data, '$.nextBillingDate') as date) as Next_Billing_Date
            ,JSON_VALUE(bt.data, '$.nextBillingPeriodAmount') as Next_Billing_Amount
            ,cast(JSON_VALUE(bt.data, '$.paidThroughDate') as date) as Paid_Through_Date
        ,JSON_VALUE(bt.data, '$.price') as Price
            ,cast(JSON_VALUE(bt.data, '$.billingPeriodStartDate')as date) as Billing_StartDate
            ,cast(JSON_VALUE(bt.data, '$.billingPeriodEndDate') as date) as Billing_EndDate
            ,JSON_VALUE(bt.data, '$.email') as Email
            ,JSON_VALUE(bt.data, '$.paymentStatus') as Payment_Status
            ,JSON_VALUE(bt.data, '$.paymentToken') as Payment_Token
            ,JSON_VALUE(bt.data, '$.processorResponseText') as Processor_Response_Text
            ,JSON_VALUE(bt.data, '$.processorResponseType') as Processor_Response_Type
            ,JSON_VALUE(bt.data, '$.numberOfTransactions') as Number_Of_Transactions
            ,JSON_VALUE(bt.data, '$.numberOfAddOns') as Number_Of_AddOns
            ,JSON_VALUE(bt.data, '$.numberOfDiscounts') as Number_Of_Discounts
            ,JSON_VALUE(bt.data, '$.isPaymentValid') as If_Payment_Valid
            ,JSON_VALUE(bt.data, '$.zombie') as If_Zombie
        from 
        [csidb].[csi].[WebhookLog] as bt                                                         /*Braintree Subscription Data*/
            join                                                                 
          (select 
              JSON_VALUE(data, '$.userID') as UserID
              ,JSON_VALUE(data, '$.subscriptionID') as SubID
              ,max(id) as max_id
          from 
              [csidb].[csi].[WebhookLog] as w
          group by 
              JSON_VALUE(data, '$.userID')
              ,JSON_VALUE(data, '$.subscriptionID')) as sub
          on 
              JSON_VALUE(bt.data, '$.subscriptionID')=sub.SubID 
              and JSON_VALUE(bt.data, '$.userID')=sub.UserID 
              and bt.id=sub.max_id
          ) as BT
          on s.userID = BT.User_ID and BT.Braintree_ID= (case when Auditlog.AuditID is null then JSON_VALUE(s.data, '$.braintreeID') else Auditlog.BraintreeID end) 
WHERE
         s.userID is not null 
         order by s.userID
